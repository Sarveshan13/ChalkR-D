Array Filter Tutorial

To check whether the integer in an array is above 45 or not between 5 elements. Click the below given button.

The driver code for the above program is given below.

function filter()
{
  var ages=[];
  var size=5;
  
  for(var i=0;i<size;i++)
  {
     ages[i]=prompt('Enter Element ' + (i+1));
  }

  function checknum(age)
  {
     return age>=45
  }
  
  document.getElementById("demo").innerHTML = ages.filter(checknum);
  }

To check whether the number of syllable of the word in an array is more than 4 words or not between 5 elements. Click the below given button.

The driver code for the above program is given below.

function filter()
{
  var words=[];
  var numw=5;
  
  for(var i=0;i<numw;i++)
  {
     words[i]=prompt('Enter Element ' + (i+1));
  }

  function checklen(syl)
  {
     return syl.length>=4;
  }
  
  document.getElementById("demo2").innerHTML = words.filter(checklen);
  }
